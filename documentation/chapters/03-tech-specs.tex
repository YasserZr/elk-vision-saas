\section{Justifications}
\begin{itemize}
    \item \textbf{Django}: Batteries-included framework with built-in auth, ORM, and excellent WebSocket support via Channels
    \item \textbf{Next.js}: React framework with App Router, SSR, and optimized performance
    \item \textbf{Multi-DB}: Elasticsearch for search, PostgreSQL for ACID transactions, MongoDB for flexible documents, Redis for real-time
    \item \textbf{Docker}: Consistent environments and easy horizontal scaling
\end{itemize}

\section{System Requirements}
\textbf{Minimum}: 4 CPU cores, 8 GB RAM, 50 GB disk.
\textbf{Recommended}: 8+ cores, 16 GB RAM, 100 GB SSD, Ubuntu 22.04+.
