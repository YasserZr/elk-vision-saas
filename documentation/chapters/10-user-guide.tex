\chapter{User Guide}

\section{Getting Started}

\subsection{Creating an Account}

\begin{enumerate}
    \item Navigate to \code{http://localhost:3000/register}
    \item Fill in the registration form:
    \begin{itemize}
        \item Username (unique identifier)
        \item Email address
        \item Password (minimum 8 characters)
    \end{itemize}
    \item Click "Sign Up" to create your account
    \item You will be redirected to the login page
\end{enumerate}

\subsection{Logging In}

\begin{enumerate}
    \item Navigate to \code{http://localhost:3000/login}
    \item Enter your username and password
    \item Click "Login" to access the dashboard
\end{enumerate}

\section{Uploading Logs}

\subsection{File Upload Interface}

\begin{figure}[H]
    \centering
    \includegraphics[width=\textwidth]{File-upload-view.png}
    \caption{Log File Upload Interface}
    \label{fig:file-upload}
\end{figure}

\subsubsection{Supported File Formats}

\begin{itemize}
    \item \textbf{JSON}: Structured log entries (one per line or array)
    \item \textbf{CSV}: Comma-separated values with headers
    \item \textbf{TXT}: Plain text log files with regex patterns
\end{itemize}

\subsubsection{Upload Steps}

\begin{enumerate}
    \item Click "Upload Logs" in the navigation menu
    \item Drag and drop your log file or click "Browse"
    \item Configure parsing options (source, environment)
    \item Click "Upload" to start processing
    \item Monitor upload progress in real-time
\end{enumerate}

\section{Searching Logs}

\subsection{Search Interface}

\begin{figure}[H]
    \centering
    \includegraphics[width=\textwidth]{Search-Filter-View.png}
    \caption{Log Search and Filter Interface}
    \label{fig:search-filter}
\end{figure}

\subsection{Search Filters}

\begin{description}
    \item[Query] Full-text search across all log fields
    \item[Level] Filter by severity (debug, info, warning, error, critical)
    \item[Source] Filter by source application
    \item[Date Range] Filter by time period (1h, 24h, 7d, custom)
    \item[Environment] Filter by environment (dev, staging, production)
\end{description}

\subsection{Search Tips}

\begin{itemize}
    \item Use quotes for exact phrases: \code{"connection refused"}
    \item Use wildcards: \code{database*}
    \item Combine filters for precise results
    \item Save frequently used searches
\end{itemize}

\subsection{Features}

\begin{itemize}
    \item \textbf{Sub-second latency}: Logs appear within 1 second of ingestion
    \item \textbf{Real-time filters}: Filter stream by level and source
    \item \textbf{Pause/Resume}: Pause auto-scroll with Spacebar
    \item \textbf{Live metrics}: View logs/second and error rate
    \item \textbf{Connection status}: WebSocket connection indicator
\end{itemize}
